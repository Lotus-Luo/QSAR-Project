# ========== Metrics ==========
def calculate_metrics(y_true: np.ndarray, y_pred: np.ndarray, y_proba: np.ndarray,
                     task: str = "classification", ef_percentile: float = 1.0) -> Dict[str, float]:
    metrics = {}
    if task == "classification":
        metrics['AUC'] = float(roc_auc_score(y_true, y_proba))
        metrics['PR_AUC'] = float(average_precision_score(y_true, y_proba))
        metrics['ACC'] = float(accuracy_score(y_true, y_pred))
        metrics['F1'] = float(f1_score(y_true, y_pred))
        metrics['MCC'] = float(matthews_corrcoef(y_true, y_pred))
        
        # EF calculation with ceil to ensure >= target percentage
        n_total = len(y_true)
        n_percentile = max(1, int(np.ceil(n_total * ef_percentile / 100.0)))
        actual_percentile = (n_percentile / n_total) * 100.0
        
        if n_percentile > 0:
            sorted_indices = np.argsort(y_proba)[::-1][:n_percentile]
            n_actives_top = np.sum(y_true[sorted_indices])
            n_actives_total = np.sum(y_true)
            
            if n_actives_total > 0:
                n_expected = n_actives_total * (n_percentile / n_total)
                metrics[f'EF{ef_percentile:g}%'] = float(n_actives_top / n_expected if n_expected > 0 else 0.0)
                metrics[f'Hit_Rate_{ef_percentile:g}%'] = float(n_actives_top / n_percentile)
            else:
                metrics[f'EF{ef_percentile:g}%'] = 0.0
                metrics[f'Hit_Rate_{ef_percentile:g}%'] = 0.0
            metrics[f'EF{ef_percentile:g}%_actual_percentile'] = float(actual_percentile)
        
        precisions, recalls, thresholds = precision_recall_curve(y_true, y_proba)
        metrics['Precision_at_0.5'] = float(precisions[np.argmin(np.abs(thresholds - 0.5))] if len(thresholds) > 0 else precisions[0])
        metrics['Recall_at_0.5'] = float(recalls[np.argmin(np.abs(thresholds - 0.5))] if len(thresholds) > 0 else recalls[0])
    else:
        metrics['R2'] = float(r2_score(y_true, y_proba))
        metrics['RMSE'] = float(math.sqrt(mean_squared_error(y_true, y_proba)))
        metrics['MAE'] = float(mean_absolute_error(y_true, y_proba))
    return metrics

def calculate_ad(X_train: np.ndarray, X_test: np.ndarray, method: str = "leverage",
                 threshold: float = 3.0, k: int = 5) -> np.ndarray:
    from scipy.spatial.distance import cdist
    n_train, n_test = X_train.shape[0], X_test.shape[0]
    
    if method == "leverage":
        try:
            scaler = StandardScaler()
            X_train_scaled = scaler.fit_transform(X_train)
            X_test_scaled = scaler.transform(X_test)
            X_pinv = np.linalg.pinv(X_train_scaled)
            h_test = np.sum((X_test_scaled @ X_pinv) ** 2, axis=1)
            h_star = 3 * (n_train + 1) / n_train
            return h_test < threshold * h_star
        except Exception as e:
            print(f"Warning: Leverage calculation failed ({e}). Using distance method.")
            return calculate_ad(X_train, X_test, method="distance", threshold=threshold)
    
    elif method == "distance":
        dist_matrix = cdist(X_test, X_train, metric='euclidean')
        min_distances = np.min(dist_matrix, axis=1)
        dist_train = cdist(X_train, X_train, metric='euclidean')
        np.fill_diagonal(dist_train, np.inf)
        mean_dist_train = np.mean(np.min(dist_train, axis=1))
        return min_distances < threshold * mean_dist_train
    
    elif method == "density":
        dist_matrix = cdist(X_train, X_test, metric='euclidean')
        sorted_distances = np.sort(dist_matrix, axis=0)
        k_distances = sorted_distances[k, :] if k < n_train else sorted_distances[-1, :]
        dist_train = cdist(X_train, X_train, metric='euclidean')
        np.fill_diagonal(dist_train, np.inf)
        sorted_train_distances = np.sort(dist_train, axis=0)
        k_distances_train = sorted_train_distances[k, :] if k < n_train else sorted_train_distances[-1, :]
        mean_k_dist_train = np.mean(k_distances_train)
        return k_distances < threshold * mean_k_dist_train
    
    raise ValueError(f"Unknown AD method: {method}")

# ========== Scaffold Split ==========
def get_scaffold(smiles: str) -> str:
    try:
        mol = Chem.MolFromSmiles(smiles)
        if mol is None: return ""
        from rdkit.Chem.Scaffolds import MurckoScaffold
        scaffold = MurckoScaffold.GetScaffoldForMol(mol)
        if scaffold is None: return ""
        return Chem.MolToSmiles(scaffold)
    except:
        return ""

class ScaffoldKFold:
    def __init__(self, n_splits: int = 5, shuffle: bool = False, random_state: int = None):
        if n_splits < 2: raise ValueError(f"n_splits must be at least 2, got {n_splits}")
        self.n_splits = n_splits
        self.shuffle = shuffle
        self.random_state = random_state

    def split(self, smiles_list: List[str], y: np.ndarray = None):
        if self.random_state is not None: np.random.seed(self.random_state)
        
        scaffolds = [get_scaffold(s) for s in smiles_list]
        scaffold_less_indices = [i for i, s in enumerate(scaffolds) if s == ""]
        
        scaffold_dict = {}
        for idx, scaffold in enumerate(scaffolds):
            if scaffold == "": continue
            if scaffold not in scaffold_dict: scaffold_dict[scaffold] = []
            scaffold_dict[scaffold].append(idx)
        
        scaffold_sizes = {s: len(i) for s, i in scaffold_dict.items()}
        sorted_scaffolds = sorted(scaffold_dict.keys(), key=lambda x: scaffold_sizes[x], reverse=True)
        
        if len(sorted_scaffolds) < self.n_splits:
            raise ValueError(f"Unique scaffolds ({len(sorted_scaffolds)}) < n_splits ({self.n_splits})")
        
        if self.shuffle:
            size_groups = {}
            for s in sorted_scaffolds:
                sz = scaffold_sizes[s]
                if sz not in size_groups: size_groups[sz] = []
                size_groups[sz].append(s)
            for sz in size_groups: np.random.shuffle(size_groups[sz])
            sorted_scaffolds = []
            for sz in sorted(size_groups.keys(), reverse=True):
                sorted_scaffolds.extend(size_groups[sz])
            np.random.shuffle(scaffold_less_indices)
        
        fold_scaffolds = [[] for _ in range(self.n_splits)]
        
        # Stratified round-robin allocation
        if len(sorted_scaffolds) >= 3:
            n_large = max(1, int(len(sorted_scaffolds) * 0.2))
            n_small = max(1, int(len(sorted_scaffolds) * 0.2))
            n_medium = len(sorted_scaffolds) - n_large - n_small
            tiers = [sorted_scaffolds[:n_large], sorted_scaffolds[n_large:n_large+n_medium], sorted_scaffolds[n_large+n_medium:]]
            for tier in tiers:
                fold_idx = 0
                for s in tier:
                    fold_scaffolds[fold_idx].append(s)
                    fold_idx = (fold_idx + 1) % self.n_splits
        else:
            fold_idx = 0
            for s in sorted_scaffolds:
                fold_scaffolds[fold_idx].append(s)
                fold_idx = (fold_idx + 1) % self.n_splits
        
        # Distribute scaffold-less molecules
        for idx in scaffold_less_indices:
            fold_sizes = [sum(scaffold_sizes[s] for s in fs) for fs in fold_scaffolds]
            min_fold = np.argmin(fold_sizes)
            special = f"_scaffold_less_{min_fold}"
            if special not in scaffold_sizes: scaffold_sizes[special] = 0
            scaffold_sizes[special] += 1
            fold_scaffolds[min_fold].append(special)
            if special not in scaffold_dict: scaffold_dict[special] = []
            scaffold_dict[special].append(idx)
        
        for fold_idx in range(self.n_splits):
            test_indices = [i for s in fold_scaffolds[fold_idx] for i in scaffold_dict[s]]
            train_indices = [i for i in range(self.n_splits) if i != fold_idx for s in fold_scaffolds[i] for i in scaffold_dict[s]]
            if len(test_indices) == 0:
                raise ValueError(f"Fold {fold_idx + 1} has empty test set")
            yield np.array(train_indices, dtype=int), np.array(test_indices, dtype=int)
    
    def get_n_splits(self): return self.n_splits

def scaffold_split(smiles_list: List[str], y: np.ndarray, test_size: float = 0.2, seed: int = 42):
    np.random.seed(seed)
    scaffolds = [get_scaffold(s) for s in smiles_list]
    scaffold_dict = {}
    for idx, scaffold in enumerate(scaffolds):
        if scaffold not in scaffold_dict: scaffold_dict[scaffold] = []
        scaffold_dict[scaffold].append(idx)
    scaffold_sizes = {s: len(i) for s, i in scaffold_dict.items()}
    sorted_scaffolds = sorted(scaffold_sizes.keys(), key=lambda x: scaffold_sizes[x], reverse=True)
    
    n_samples = len(smiles_list)
    n_test_target = int(n_samples * test_size)
    
    best_train, best_test, best_dev = None, None, float('inf')
    for iter in range(10):
        np.random.seed(seed + iter)
        np.random.shuffle(sorted_scaffolds)
        train_i, test_i = [], []
        test_size_curr = 0
        for s in sorted_scaffolds:
            sz = scaffold_sizes[s]
            if test_size_curr + sz <= n_test_target * 1.05:
                test_i.extend(scaffold_dict[s])
                test_size_curr += sz
            else:
                train_i.extend(scaffold_dict[s])
        dev = abs(len(test_i) - n_test_target) / n_test_target
        if dev < best_dev:
            best_dev = dev
            best_train = train_i.copy()
            best_test = test_i.copy()
    
    return np.array(best_train, dtype=int), np.array(best_test, dtype=int)

# ========== PyTorch Models ==========
class MLPClassifier(nn.Module):
    def __init__(self, input_dim: int, hidden_dims: List[int] = [512, 256], dropout: float = 0.3):
        super().__init__()
        layers = []
        prev = input_dim
        for hd in hidden_dims:
            layers.extend([nn.Linear(prev, hd), nn.BatchNorm1d(hd), nn.ReLU(), nn.Dropout(dropout)])
            prev = hd
        layers.append(nn.Linear(prev, 1))
        self.network = nn.Sequential(*layers)
    def forward(self, x): return self.network(x)

class MLPRegressor(nn.Module):
    def __init__(self, input_dim: int, hidden_dims: List[int] = [512, 256], dropout: float = 0.3):
        super().__init__()
        layers = []
        prev = input_dim
        for hd in hidden_dims:
            layers.extend([nn.Linear(prev, hd), nn.BatchNorm1d(hd), nn.ReLU(), nn.Dropout(dropout)])
            prev = hd
        layers.append(nn.Linear(prev, 1))
        self.network = nn.Sequential(*layers)
    def forward(self, x): return self.network(x)

def smiles_to_graph(smiles: str) -> Optional[Data]:
    if not GAT_AVAILABLE: return None
    try:
        mol = Chem.MolFromSmiles(smiles)
        if mol is None: return None
        mol = Chem.AddHs(mol)
        num_atoms = mol.GetNumAtoms()
        node_features = []
        for atom in mol.GetAtoms():
            features = [atom.GetAtomicNum(), atom.GetDegree(), atom.GetFormalCharge(),
                       atom.GetHybridization().real, atom.GetIsAromatic(), atom.GetTotalNumHs()]
            node_features.append(features)
        x = torch.tensor(node_features, dtype=torch.float)
        edge_indices = []
        for bond in mol.GetBonds():
            i, j = bond.GetBeginAtomIdx(), bond.GetEndAtomIdx()
            edge_indices.extend([[i, j], [j, i]])
        edge_index = torch.tensor(edge_indices, dtype=torch.long).t().contiguous() if edge_indices else torch.empty((2, 0), dtype=torch.long)
        return Data(x=x, edge_index=edge_index)
    except:
        return None

class MoleculeDataset(Dataset):
    def __init__(self, smiles_list, labels):
        self.graphs = []
        for smiles in tqdm(smiles_list):
            graph = smiles_to_graph(smiles)
            self.graphs.append(graph)
        valid_indices = [i for i, g in enumerate(self.graphs) if g is not None]
        self.valid_indices = valid_indices
        self.graphs = [self.graphs[i] for i in valid_indices]
        self.labels = [labels[i] for i in valid_indices]
        self.smiles_list = [smiles_list[i] for i in valid_indices]
        print(f"Successfully converted {len(self.graphs)}/{len(smiles_list)} molecules")
    def __len__(self): return len(self.graphs)
    def __getitem__(self, idx):
        graph = self.graphs[idx]
        graph.y = torch.tensor([self.labels[idx]], dtype=torch.float32)
        return graph

class ChemBERTaModel(nn.Module):
    def __init__(self, model_name: str = "DeepChem/ChemBERTa-77M-MLM", num_labels: int = 1):
        super().__init__()
        mapping = {
            "DeepChem/ChemBERTa-77M-MLM": "chemberta_77m_mlm",
            "DeepChem/ChemBERTa-77M-MTR": "chemberta_77m_mtr",
            "seyonec/ChemBERTa-zinc-base-v1": "chemberta_zinc_v1",
        }
        local_folder = None
        for k, v in mapping.items():
            if k.lower() == model_name.lower():
                local_folder = v
                break
        if local_folder is None:
            local_folder = model_name.split('/')[-1].replace("-", "_")
        local_path = os.path.join("pretrained_model", "all_chemberta_models", local_folder)
        
        if os.path.exists(local_path) and os.path.exists(os.path.join(local_path, "config.json")):
            load_path, use_local = local_path, True
            print(f"[INFO] Loading ChemBERTa from LOCAL: {load_path}")
        else:
            load_path, use_local = model_name, False
            print(f"[INFO] Local model not found. Downloading from HF: {load_path}")
        
        self.tokenizer = AutoTokenizer.from_pretrained(load_path, local_files_only=use_local)
        self.model = AutoModelForSequenceClassification.from_pretrained(
            load_path, num_labels=num_labels, ignore_mismatched_sizes=True, local_files_only=use_local
        )
    def forward(self, input_ids, attention_mask):
        outputs = self.model(input_ids=input_ids, attention_mask=attention_mask)
        return outputs.logits.squeeze(-1)

class ChemBERTaDataset(Dataset):
    def __init__(self, smiles_list, labels, tokenizer, max_length=128, logger=None):
        self.smiles_list = smiles_list
        self.labels = labels
        self.tokenizer = tokenizer
        self.max_length = max_length
    def __len__(self): return len(self.smiles_list)
    def __getitem__(self, idx):
        smiles = self.smiles_list[idx]
        label = self.labels[idx]
        if not smiles or smiles.strip() == "":
            return None
        encoding = self.tokenizer(smiles, truncation=True, padding='max_length',
                                 max_length=self.max_length, return_tensors='pt')
        return {
            'input_ids': encoding['input_ids'].squeeze(0),
            'attention_mask': encoding['attention_mask'].squeeze(0),
            'label': torch.tensor(label, dtype=torch.float32)
        }

def chemberta_collate_fn(batch):
    valid_batch = [item for item in batch if item is not None]
    if not valid_batch:
        return {
            'input_ids': torch.zeros((0, 128), dtype=torch.long),
            'attention_mask': torch.zeros((0, 128), dtype=torch.long),
            'label': torch.zeros((0,), dtype=torch.float32)
        }
    return {
        'input_ids': torch.stack([item['input_ids'] for item in valid_batch]),
        'attention_mask': torch.stack([item['attention_mask'] for item in valid_batch]),
        'label': torch.stack([item['label'] for item in valid_batch])
    }

                    setattr(self, 'num_layers', num_layers),
                    setattr(self, 'convs', nn.ModuleList([
                        GATConv(num_node_features if i == 0 else hidden_dim * num_heads, hidden_dim, heads=num_heads, dropout=dropout)
                        for i in range(num_layers)
                    ])),
                    setattr(self, 'output_dim', hidden_dim * num_heads),
                    setattr(self, 'fc', nn.Linear(hidden_dim * num_heads, 1)),
                    setattr(self, 'dropout', nn.Dropout(dropout))
                )[0],
                'forward': lambda self, x, edge_index, batch: (
                    [self.convs[i](x, edge_index) for i in range(self.num_layers)],
                    (x := F.elu(x)),
                    (x := self.dropout(x)),
                    (x := global_mean_pool(x, batch)),
                    self.fc(x).squeeze(dim=-1)
                )[-1]
            }),
            'params': {'num_node_features': 6, 'hidden_dim': 64, 'num_heads': 4, 'num_layers': 3, 'dropout': 0.3}
        }
    
    if CHEMBERTA_AVAILABLE:
        registry['ChemBERTa'] = {
            'type': 'transformer',
            'model_class': ChemBERTaModel,
            'params': {'model_name': 'DeepChem/ChemBERTa-77M-MLM', 'num_labels': 1}
        }
    
    return registry

# ========== Training Functions ==========
def train_pytorch_model(model, train_loader, val_loader, config: QSARConfig,
                       task: str = "classification", logger: logging.Logger = None,
                       model_type: str = "default") -> nn.Module:
    if logger is None: logger = logging.getLogger(__name__)
    device = DEVICE
    model = model.to(device)
    
    if model_type == "transformer":
        optimizer = torch.optim.AdamW(model.parameters(), lr=5e-5, weight_decay=0.01)
        scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=config.max_epochs, eta_min=1e-7)
    else:
        optimizer = torch.optim.Adam(model.parameters(), lr=config.learning_rate, weight_decay=1e-5)
        scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(optimizer, mode='min', factor=0.5, patience=5)
    
    criterion = nn.BCEWithLogitsLoss() if task == "classification" else nn.MSELoss()
    
    best_val_loss, patience_counter, best_model_state, best_epoch = float('inf'), 0, None, 0
    
    for epoch in range(config.max_epochs):
        model.train()
        train_loss = 0.0
        
        pbar = tqdm(train_loader, desc=f"Epoch {epoch+1}/{config.max_epochs} [Train]", leave=False, disable=logger is None)
        for batch in pbar:
            optimizer.zero_grad()
            if isinstance(batch, Data):
                batch = batch.to(device)
                labels = batch.y
                outputs = model(batch.x, batch.edge_index, batch.batch)
            elif isinstance(batch, (list, tuple)) and len(batch) == 2:
                inputs, labels = batch
                inputs, labels = inputs.to(device), labels.to(device)
                outputs = model(inputs).squeeze(-1)
            else:
                input_ids, attention_mask, labels = batch['input_ids'].to(device), batch['attention_mask'].to(device), batch['label'].to(device)
                outputs = model(input_ids, attention_mask)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()
            train_loss += loss.item()
            pbar.set_postfix({'loss': f'{loss.item():.4f}'})
        pbar.close()
        train_loss /= len(train_loader)
        
        if torch.cuda.is_available(): torch.cuda.empty_cache()
        
        if val_loader is None:
            val_loss = train_loss
            logger.info(f"Epoch {epoch+1}/{config.max_epochs} - Train Loss: {train_loss:.4f} (No validation)")
            if model_type == "transformer": scheduler.step()
            if train_loss < best_val_loss:
                best_val_loss, patience_counter, best_model_state, best_epoch = train_loss, 0, copy.deepcopy(model.state_dict()), epoch + 1
        else:
            model.eval()
            val_loss, all_preds, all_labels = 0.0, [], []
            with torch.no_grad():
                pbar_val = tqdm(val_loader, desc=f"Epoch {epoch+1}/{config.max_epochs} [Val]", leave=False, disable=logger is None)
                for batch in pbar_val:
                    if isinstance(batch, Data):
                        batch = batch.to(device)
                        labels = batch.y
                        outputs = model(batch.x, batch.edge_index, batch.batch)
                    elif isinstance(batch, (list, tuple)) and len(batch) == 2:
                        inputs, labels = batch
                        inputs, labels = inputs.to(device), labels.to(device)
                        outputs = model(inputs).squeeze(-1)
                    else:
                        input_ids, attention_mask, labels = batch['input_ids'].to(device), batch['attention_mask'].to(device), batch['label'].to(device)
                        outputs = model(input_ids, attention_mask)
                    loss = criterion(outputs, labels)
                    val_loss += loss.item()
                    all_preds.extend(outputs.cpu().numpy())
                    all_labels.extend(labels.cpu().numpy())
                    pbar_val.set_postfix({'loss': f'{loss.item():.4f}'})
                pbar_val.close()
            val_loss /= len(val_loader)
            
            if task == "classification":
                val_preds_proba = torch.sigmoid(torch.tensor(all_preds)).numpy()
                val_preds_binary = (val_preds_proba >= 0.5).astype(int)
                val_auc = float(roc_auc_score(all_labels, val_preds_proba))
                val_acc = float(accuracy_score(all_labels, val_preds_binary))
                logger.info(f"Epoch {epoch+1}/{config.max_epochs} - Train Loss: {train_loss:.4f}, Val Loss: {val_loss:.4f}, Val AUC: {val_auc:.4f}, Val ACC: {val_acc:.4f}")
            else:
                val_r2 = float(r2_score(all_labels, all_preds))
                logger.info(f"Epoch {epoch+1}/{config.max_epochs} - Train Loss: {train_loss:.4f}, Val Loss: {val_loss:.4f}, Val R2: {val_r2:.4f}")
            
            if model_type == "transformer":
                scheduler.step()
            else:
                scheduler.step(val_loss)
            
            if val_loss < best_val_loss:
                best_val_loss, patience_counter, best_model_state, best_epoch = val_loss, 0, copy.deepcopy(model.state_dict()), epoch + 1
            else:
                patience_counter += 1
                if patience_counter >= config.early_stopping_patience:
                    logger.info(f"Early stopping at epoch {epoch+1}")
                    break
    
    if best_model_state is not None:
        model.load_state_dict(best_model_state)
        logger.info(f"Loaded best model weights from epoch {best_epoch}")
    
    return model

def predict_pytorch_model(model, data_loader, task: str = "classification") -> Tuple[np.ndarray, np.ndarray]:
    if data_loader is None: return None, None
    device = DEVICE
    model = model.to(device)
    model.eval()
    all_preds = []
    with torch.no_grad():
        for batch in data_loader:
            if isinstance(batch, Data):
                batch = batch.to(device)
                outputs = model(batch.x, batch.edge_index, batch.batch)
            elif isinstance(batch, (list, tuple)) and len(batch) == 2:
                inputs, _ = batch
                inputs = inputs.to(device)
                outputs = model(inputs).squeeze(-1)
            else:
                input_ids, attention_mask = batch['input_ids'].to(device), batch['attention_mask'].to(device)
                outputs = model(input_ids, attention_mask)
            all_preds.extend(outputs.cpu().numpy())
    all_preds = np.array(all_preds)
    if torch.cuda.is_available(): torch.cuda.empty_cache()
    
    if task == "classification":
        all_proba = torch.sigmoid(torch.tensor(all_preds)).numpy()
        all_labels = (all_proba >= 0.5).astype(int)
        return all_labels, all_proba
    else:
        return all_preds, all_preds

def get_model_type(model_key: str) -> str:
    gradient_based = ['LR', 'SVC', 'Ridge']
    return 'gradient_based' if model_key in gradient_based else 'tree_based'

def train_model_wrapper(model_key: str, model_config: Dict[str, Any], X_train, y_train,
                       X_val, y_val, config: QSARConfig, task: str = "classification",
                       smiles_train: Optional[List[str]] = None, smiles_val: Optional[List[str]] = None,
                       logger: logging.Logger = None) -> Dict[str, Any]:
    if logger is None: logger = logging.getLogger(__name__)
    logger.info(f"\n--- Training {model_key} ---")
    
    model_type = model_config.get('type', 'sklearn')
    
    if model_type == 'sklearn':
        model = model_config['model'](**model_config['params'])
        model.fit(X_train, y_train)
        
        if hasattr(model, 'predict_proba'):
            y_proba_train = model.predict_proba(X_train)[:, 1]
        elif hasattr(model, 'decision_function'):
            y_proba_train = model.decision_function(X_train)
            if task == "classification":
                y_proba_train = 1.0 / (1.0 + np.exp(-y_proba_train))
        else:
            y_proba_train = model.predict(X_train)
        y_pred_train = (y_proba_train >= 0.5).astype(int) if task == "classification" else y_proba_train
        
        y_proba_val, y_pred_val = None, None
        if X_val is not None and len(X_val) > 0:
            if hasattr(model, 'predict_proba'):
                y_proba_val = model.predict_proba(X_val)[:, 1]
            elif hasattr(model, 'decision_function'):
                y_proba_val = model.decision_function(X_val)
                if task == "classification":
                    y_proba_val = 1.0 / (1.0 + np.exp(-y_proba_val))
            else:
                y_proba_val = model.predict(X_val)
            y_pred_val = (y_proba_val >= 0.5).astype(int) if task == "classification" else y_proba_val
        
        return {
            'model': model, 'y_pred_train': y_pred_train, 'y_proba_train': y_proba_train,
            'y_pred_val': y_pred_val, 'y_proba_val': y_proba_val
        }
    
    elif model_type == 'pytorch':
        if X_train is not None:
            actual_input_dim = X_train.shape[1]
            model_config = model_config.copy()
            model_config['params'] = model_config['params'].copy()
            model_config['params']['input_dim'] = actual_input_dim
        
        model = model_config['model_class'](**model_config['params'])
        train_dataset = TensorDataset(torch.FloatTensor(X_train), torch.FloatTensor(y_train))
        train_loader = DataLoader(train_dataset, batch_size=config.batch_size, shuffle=True)
        
        val_loader = None
        if X_val is not None and len(X_val) > 0:
            val_dataset = TensorDataset(torch.FloatTensor(X_val), torch.FloatTensor(y_val))
            val_loader = DataLoader(val_dataset, batch_size=config.batch_size, shuffle=False)
        
        model = train_pytorch_model(model, train_loader, val_loader, config, task, logger, model_type="pytorch")
        y_pred_train, y_proba_train = predict_pytorch_model(model, train_loader, task)
        y_pred_val, y_proba_val = predict_pytorch_model(model, val_loader, task) if val_loader is not None else (None, None)
        
        return {
            'model': model, 'y_pred_train': y_pred_train, 'y_proba_train': y_proba_train,
            'y_pred_val': y_pred_val, 'y_proba_val': y_proba_val
        }
    
    elif model_type == 'pytorch_geometric':
        if smiles_train is None:
            logger.error(f"SMILES data required for {model_key}")
            return None
        
        model = model_config['model_class'](**model_config['params'])
        train_dataset = MoleculeDataset(smiles_train, y_train)
        train_loader = GeometricDataLoader(train_dataset, batch_size=config.batch_size, shuffle=True)
        y_train_filtered = y_train[train_dataset.valid_indices]
        
        val_loader, y_val_filtered = None, None
        if smiles_val is not None and len(smiles_val) > 0:
            val_dataset = MoleculeDataset(smiles_val, y_val)
            val_loader = GeometricDataLoader(val_dataset, batch_size=config.batch_size, shuffle=False)
            y_val_filtered = y_val[val_dataset.valid_indices]
        
        model = train_pytorch_model(model, train_loader, val_loader, config, task, logger, model_type="pytorch_geometric")
        y_pred_train, y_proba_train = predict_pytorch_model(model, train_loader, task)
        y_pred_val, y_proba_val = predict_pytorch_model(model, val_loader, task) if val_loader is not None else (None, None)
        
        return {
            'model': model, 'y_pred_train': y_pred_train, 'y_proba_train': y_proba_train,
            'y_pred_val': y_pred_val, 'y_proba_val': y_proba_val,
            'valid_indices_train': train_dataset.valid_indices,
            'valid_indices_val': val_dataset.valid_indices if val_loader is not None else None
        }
    
    elif model_type == 'transformer':
        if smiles_train is None:
            logger.error(f"SMILES data required for {model_key}")
            return None
        
        model = model_config['model_class'](**model_config['params'])
        train_dataset = ChemBERTaDataset(smiles_train, y_train, model.tokenizer)
        train_loader = DataLoader(train_dataset, batch_size=config.batch_size, shuffle=True, collate_fn=chemberta_collate_fn)
        
        val_loader = None
        if smiles_val is not None and len(smiles_val) > 0:
            val_dataset = ChemBERTaDataset(smiles_val, y_val, model.tokenizer)
            val_loader = DataLoader(val_dataset, batch_size=config.batch_size, shuffle=False, collate_fn=chemberta_collate_fn)
        
        model = train_pytorch_model(model, train_loader, val_loader, config, task, logger, model_type="transformer")
        y_pred_train, y_proba_train = predict_pytorch_model(model, train_loader, task)
        y_pred_val, y_proba_val = predict_pytorch_model(model, val_loader, task) if val_loader is not None else (None, None)
        
        return {
            'model': model, 'y_pred_train': y_pred_train, 'y_proba_train': y_proba_train,
            'y_pred_val': y_pred_val, 'y_proba_val': y_proba_val
        }
    
    else:
        logger.error(f"Unknown model type: {model_type}")
        return None

# ========== Feature Processing ==========
def apply_global_feature_filtering(X: np.ndarray, config: QSARConfig, logger: logging.Logger = None,
                                   feature_names: Optional[List[str]] = None):
    X_filtered = X.copy()
    feature_mask = np.ones(X.shape[1], dtype=bool)
    
    if config.min_frequency > 0 and np.all(np.isin(X, [0, 1])):
        n_samples = X.shape[0]
        min_count = int(n_samples * config.min_frequency)
        feature_counts = np.sum(X, axis=0)
        feature_mask = (feature_counts >= min_count) & (feature_counts <= n_samples - min_count)
        X_filtered = X_filtered[:, feature_mask]
        if logger:
            logger.info(f"Global feature filtering (frequency): {X.shape[1]} → {X_filtered.shape[1]} features")
    elif config.variance_threshold > 0:
        selector = VarianceThreshold(threshold=config.variance_threshold)
        X_filtered = selector.fit_transform(X_filtered)
        feature_mask = selector.get_support()
        if logger:
            logger.info(f"Global feature filtering (variance): {X.shape[1]} → {X_filtered.shape[1]} features")
    else:
        if logger: logger.info(f"Global feature filtering: skipped")
    
    feature_names_filtered = None
    if feature_names is not None:
        feature_names_filtered = [name for name, keep in zip(feature_names, feature_mask) if keep]
    
    return X_filtered, feature_mask, feature_names_filtered

def apply_feature_processing(X_train, X_val, config: QSARConfig, logger: logging.Logger = None, model_type='none'):
    if model_type == 'gradient_based':
        scaler = StandardScaler()
        X_train_processed = scaler.fit_transform(X_train)
        X_val_processed = scaler.transform(X_val) if X_val is not None else None
        return X_train_processed, X_val_processed, np.ones(X_train.shape[1], dtype=bool), scaler, np.ones(X_train.shape[1], dtype=bool), None
    else:
        return X_train, X_val, np.ones(X_train.shape[1], dtype=bool), None, np.ones(X_train.shape[1], dtype=bool), None

# ========== Fingerprint Generation ==========
def generate_fingerprints_from_csv(df: pd.DataFrame, smiles_column: str, fp_types: List[str],
                                  logger: logging.Logger = None) -> pd.DataFrame:
    from rdkit import Chem
    from rdkit.Chem import AllChem, MACCSkeys, rdMolDescriptors
    
    df = df.copy()
    smiles_list = df[smiles_column].tolist()
    
    for fp_type in fp_types:
        if fp_type == "morgan":
            fingerprints = []
            for smiles in tqdm(smiles_list, desc=f"Generating {fp_type} fingerprints"):
                try:
                    mol = Chem.MolFromSmiles(smiles)
                    if mol is None:
                        fingerprints.append(np.zeros(1024))
                        continue
                    fp = AllChem.GetMorganFingerprintAsBitVect(mol, 2, nBits=1024)
                    arr = np.zeros((1,), dtype=int)
                    AllChem.DataStructs.ConvertToNumpyArray(fp, arr)
                    fingerprints.append(arr)
                except:
                    fingerprints.append(np.zeros(1024))
            fp_df = pd.DataFrame(fingerprints, columns=[f'morgan_{i}' for i in range(1024)])
            df = pd.concat([df, fp_df], axis=1)
        
        elif fp_type == "maccs":
            fingerprints = []
            for smiles in tqdm(smiles_list, desc=f"Generating {fp_type} fingerprints"):
                try:
                    mol = Chem.MolFromSmiles(smiles)
                    if mol is None:
                        fingerprints.append(np.zeros(167))
                        continue
                    fp = MACCSkeys.GenMACCSKeys(mol)
                    arr = np.zeros((1,), dtype=int)
                    AllChem.DataStructs.ConvertToNumpyArray(fp, arr)
                    fingerprints.append(arr)
                except:
                    fingerprints.append(np.zeros(167))
            fp_df = pd.DataFrame(fingerprints, columns=[f'maccs_{i}' for i in range(167)])
            df = pd.concat([df, fp_df], axis=1)
        
        elif fp_type == "rdkit":
            fingerprints = []
            for smiles in tqdm(smiles_list, desc=f"Generating {fp_type} fingerprints"):
                try:
                    mol = Chem.MolFromSmiles(smiles)
                    if mol is None:
                        fingerprints.append(np.zeros(2048))
                        continue
                    fp = Chem.RDKFingerprint(mol)
                    arr = np.zeros((1,), dtype=int)
                    AllChem.DataStructs.ConvertToNumpyArray(fp, arr)
                    fingerprints.append(arr)
                except:
                    fingerprints.append(np.zeros(2048))
            fp_df = pd.DataFrame(fingerprints, columns=[f'rdkit_{i}' for i in range(2048)])
            df = pd.concat([df, fp_df], axis=1)
    
    if logger: logger.info(f"Fingerprint generation complete. Total columns: {df.shape[1]}")
    return df

# ========== Validation ==========
def validate_smiles(smiles_list: List[str], logger: logging.Logger = None) -> Tuple[List[bool], List[int]]:
    from rdkit import Chem
    valid = []
    for smiles in smiles_list:
        try:
            mol = Chem.MolFromSmiles(smiles)
            valid.append(mol is not None)
        except:
            valid.append(False)
    invalid_indices = [i for i, v in enumerate(valid) if not v]
    if logger and invalid_indices:
        logger.warning(f"Found {len(invalid_indices)} invalid SMILES out of {len(smiles_list)}")
    return valid, invalid_indices

def check_models_require_fingerprints(selected_models: List[str]) -> bool:
    fp_models = ['LR', 'RFC', 'SVC', 'XGBC', 'LGBMC', 'ETC', 'MLP']
    return any(m in selected_models for m in fp_models)

# ========== Train Single Fold (Multi-Seed Support) ==========
def train_single_fold(config: QSARConfig, X_train, y_train, X_val, y_val,
                     smiles_train: Optional[List[str]] = None, smiles_val: Optional[List[str]] = None,
                     ids_train: Optional[List[str]] = None, ids_val: Optional[List[str]] = None,
                     registry: Dict[str, Dict[str, Any]] = None,
                     fold_idx: Optional[int] = None,
                     save_models: bool = False,
                     model_save_dir: Optional[Path] = None,
                     logger: logging.Logger = None) -> List[Dict[str, Any]]:
    """Train models on a single train/validation split with multi-seed support"""
    if logger is None: logger = logging.getLogger(__name__)
    if registry is None:
        input_dim = X_train.shape[1] if X_train is not None else None
        registry = build_model_registry(task=config.task, input_dim=input_dim)
    
    fold_prefix = f"Fold {fold_idx} - " if fold_idx is not None else ""
    logger.info(f"\n{'='*60}")
    logger.info(f"{fold_prefix}Training samples: {len(y_train)}, Validation samples: {len(y_val)}")
    logger.info(f"{fold_prefix}Using {len(config.seeds)} seeds: {config.seeds}")
    logger.info(f"{'='*60}")
    
    # Apply feature processing (filtering + standardization)
    if X_train is not None and X_val is not None:
        logger.info(f"{fold_prefix}Applying feature processing...")
        X_train_processed, X_val_processed, feature_mask, _, _, _ = apply_feature_processing(
            X_train, X_val, config, logger, model_type='none'
        )
    else:
        X_train_processed, X_val_processed, feature_mask = X_train, X_val, None
    
    # Train each model
    results = []
    valid_models = [m for m in config.selected_models if m in registry]
    
    for model_key in valid_models:
        model_config = registry[model_key]
        model_type_category = 'deep_learning' if model_config['type'] in ['pytorch', 'pytorch_geometric', 'transformer'] else 'traditional'
        
        logger.info(f"\n--- {fold_prefix}Training {model_key} with {len(config.seeds)} seeds ---")
        
        # Prepare data based on model type
        if model_config['type'] in ['sklearn']:
            model_preprocessing_type = get_model_type(model_key)
            if X_train_processed is not None and X_val_processed is not None:
                if model_preprocessing_type == 'gradient_based':
                    from sklearn.preprocessing import StandardScaler
                    scaler = StandardScaler()
                    X_train_final = scaler.fit_transform(X_train_processed)
                    X_val_final = scaler.transform(X_val_processed)
                    logger.info(f"{fold_prefix}Standardization applied for {model_key}")
                else:
                    X_train_final, X_val_final = X_train_processed, X_val_processed
                    logger.info(f"{fold_prefix}Standardization skipped for {model_key} (tree-based)")
            else:
                X_train_final, X_val_final = None, None
        else:
            if model_config['type'] == 'pytorch':
                if X_train_processed is not None and X_val_processed is not None:
                    from sklearn.preprocessing import StandardScaler
                    scaler = StandardScaler()
                    X_train_final = scaler.fit_transform(X_train_processed)
                    X_val_final = scaler.transform(X_val_processed)
                    logger.info(f"{fold_prefix}Standardization applied for {model_key}")
                else:
                    X_train_final, X_val_final = None, None
            else:
                X_train_final, X_val_final = None, None
        
        # Multi-seed training loop
        seed_results = []
        
        for seed_idx, seed in enumerate(config.seeds):
            logger.info(f"\n  {fold_prefix}{model_key} - Seed {seed_idx+1}/{len(config.seeds)} (seed={seed})")
            set_all_seeds(seed, logger)
            
            # Prepare data based on model type
            if model_config['type'] in ['pytorch_geometric', 'transformer']:
                if smiles_train is None or smiles_val is None:
                    logger.warning(f"{fold_prefix}SMILES data not available, skipping {model_key}")
                    break
                result = train_model_wrapper(
                    model_key, model_config, None, y_train, None, y_val,
                    config, config.task, smiles_train, smiles_val, logger
                )
            else:
                result = train_model_wrapper(
                    model_key, model_config, X_train_final, y_train, X_val_final, y_val,
                    config, config.task, None, None, logger
                )
            
            if result is None:
                logger.warning(f"  {fold_prefix}{model_key} - Seed {seed} failed")
                continue
            
            # Save model if requested
            if save_models and model_save_dir is not None and 'model' in result:
                fold_str = f"fold{fold_idx}" if fold_idx is not None else "single"
                model_filename = f"{model_key}_{fold_str}_seed{seed}"
                model_path = model_save_dir / model_filename
                try:
                    if model_type_category == 'traditional':
                        _save_sklearn_model(result['model'], model_path)
                        logger.info(f"  {fold_prefix}✓ Saved model: {model_filename}.joblib")
                    else:
                        _save_pytorch_model(result['model'], model_path, model_config['type'])
                        logger.info(f"  {fold_prefix}✓ Saved model: {model_filename}.pt")
                except Exception as e:
                    logger.warning(f"  {fold_prefix}✗ Failed to save model: {e}")
            
            # Calculate metrics
            val_metrics = calculate_metrics(y_val, result['y_pred_val'], result['y_proba_val'], config.task, config.ef_percentile)
            train_metrics = calculate_metrics(y_train, result['y_pred_train'], result['y_proba_train'], config.task, config.ef_percentile)
            
            seed_result_dict = {
                'fold': fold_idx, 'model': model_key, 'model_type': model_type_category, 'seed': seed,
                **{f'{k}_val': v for k, v in val_metrics.items()},
                **{f'{k}_train': v for k, v in train_metrics.items()}
            }
            seed_results.append(seed_result_dict)
            
            # GPU memory management
            if model_type_category == 'deep_learning' and torch.cuda.is_available():
                torch.cuda.empty_cache()
                if 'model' in result: del result['model']
                torch.cuda.empty_cache()
            
            # Log metrics
            if config.task == "classification":
                logger.info(f"  {fold_prefix}✓ {model_key} - Seed {seed} completed")
                logger.info(f"    Val AUC: {val_metrics.get('AUC', 'N/A'):.4f}, PR-AUC: {val_metrics.get('PR_AUC', 'N/A'):.4f}")
                logger.info(f"    Val MCC: {val_metrics.get('MCC', 'N/A'):.4f}, EF{config.ef_percentile:g}%: {val_metrics.get(f'EF{config.ef_percentile:g}%', 'N/A'):.4f}")
            else:
                logger.info(f"  {fold_prefix}✓ {model_key} - Seed {seed} completed")
                logger.info(f"    Val R2: {val_metrics.get('R2', 'N/A'):.4f}, RMSE: {val_metrics.get('RMSE', 'N/A'):.4f}")
        
        # Skip if no seeds succeeded
        if not seed_results:
            logger.warning(f"{fold_prefix}{model_key} failed on all seeds")
            continue
        
        # Aggregate results across seeds
        logger.info(f"\n{fold_prefix}Aggregating results for {model_key} across {len(seed_results)} seeds")
        
        metric_keys = [k for k in seed_results[0].keys() if k not in ['fold', 'model', 'model_type', 'seed']]
        
        aggregated_result = {
            'fold': fold_idx, 'model': model_key, 'model_type': model_type_category, 'num_seeds': len(seed_results)
        }
        
        for metric_key in metric_keys:
            values = [r[metric_key] for r in seed_results]
            aggregated_result[f'{metric_key}_mean'] = np.mean(values)
            aggregated_result[f'{metric_key}_std'] = np.std(values)
        
        results.append(aggregated_result)
        
        # Log aggregated metrics
        if config.task == "classification":
            logger.info(f"{fold_prefix}✓ {model_key} aggregated results:")
            logger.info(f"  Val AUC: {aggregated_result.get('AUC_val_mean', 'N/A'):.4f} ± {aggregated_result.get('AUC_val_std', 'N/A'):.4f}")
            logger.info(f"  Val PR-AUC: {aggregated_result.get('PR_AUC_val_mean', 'N/A'):.4f} ± {aggregated_result.get('PR_AUC_val_std', 'N/A'):.4f}")
            logger.info(f"  Val MCC: {aggregated_result.get('MCC_val_mean', 'N/A'):.4f} ± {aggregated_result.get('MCC_val_std', 'N/A'):.4f}")
        else:
            logger.info(f"{fold_prefix}✓ {model_key} aggregated results:")
            logger.info(f"  Val R2: {aggregated_result.get('R2_val_mean', 'N/A'):.4f} ± {aggregated_result.get('R2_val_std', 'N/A'):.4f}")
            logger.info(f"  Val RMSE: {aggregated_result.get('RMSE_val_mean', 'N/A'):.4f} ± {aggregated_result.get('RMSE_val_std', 'N/A'):.4f}")
    
    return results

        format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',

    )
    logger = logging.getLogger(__name__)

    if args.config:
        config = QSARConfig.from_yaml(args.config) if args.config.endswith('.yaml') else QSARConfig.from_json(args.config)
    else:
        config = QSARConfig()

    config.input_path = args.input
    config.output_dir = args.output
    config.task = args.task
    config.label_column = args.label
    config.pic50_column = args.pic50
    config.smiles_column = args.smiles
    config.id_column = args.id
    config.test_size = args.test_size
    config.folds = args.folds
    config.seed = args.seed
    config.seeds = args.seeds
    config.selected_models = args.models
    config.split_method = args.split_method
    config.cv_split_method = args.cv_split_method
    config.ef_percentile = args.ef_percentile
    config.external_test_metric = args.external_metric
    config.variance_threshold = args.variance_threshold
    config.min_frequency = args.min_frequency
    config.max_epochs = args.max_epochs
    config.batch_size = args.batch_size
    config.learning_rate = args.learning_rate
    config.early_stopping_patience = args.early_stopping
    config.n_jobs = args.n_jobs
    config.auto_generate_fingerprints = args.generate_fp
    config.fingerprint_types = args.fp_types
    config.run_shap = args.run_shap
    
    logger.info("QSAR Batch Modeling Pipeline")
    logger.info(f"Task: {config.task}")
    logger.info(f"Input: {config.input_path}")
    logger.info(f"Output: {config.output_dir}")
    logger.info(f"Selected models: {config.selected_models}")
    
    # Load data
    df = read_table(Path(config.input_path))
    logger.info(f"Loaded data: {df.shape[0]} samples, {df.shape[1]} columns")
    
    # Validate SMILES
    smiles_list = df[config.smiles_column].tolist()
    valid, invalid_indices = validate_smiles(smiles_list, logger)
    if invalid_indices:
        logger.warning(f"Removing {len(invalid_indices)} samples with invalid SMILES")
        df = df.drop(index=invalid_indices).reset_index(drop=True)
        smiles_list = df[config.smiles_column].tolist()
    
    # Generate fingerprints if needed
    if config.auto_generate_fingerprints and check_models_require_fingerprints(config.selected_models):
        logger.info("Generating fingerprints...")
        df = generate_fingerprints_from_csv(df, config.smiles_column, config.fingerprint_types, logger)
    
    # Extract features and labels
    ids = df[config.id_column].tolist() if config.id_column in df.columns else list(range(len(df)))
    smiles_list = df[config.smiles_column].tolist()
    
    if config.task == 'classification':
        y = df[config.label_column].values.astype(int)
    else:
        y = df[config.pic50_column].values.astype(float)
    
    fp_cols = select_fp_columns(df)
    if fp_cols:
        X = df[fp_cols].values
        logger.info(f"Using {len(fp_cols)} fingerprint features")
    else:
        X = None
        logger.info("No fingerprint features found")
    
    # Create output directory
    out_dir = Path(config.output_dir)
    out_dir.mkdir(parents=True, exist_ok=True)
    for subdir in ['logs', 'results', 'models', 'feature_processors', 'predictions']:
        (out_dir / subdir).mkdir(exist_ok=True)
    
    # Save config
    config_dict = config.__dict__.copy()
    config_dict['selected_models'] = config.selected_models
    with open(out_dir / "config.json", 'w') as f:
        json.dump(config_dict, f, indent=2, default=str)
    
    # Build model registry
    input_dim = X.shape[1] if X is not None else None
    registry = build_model_registry(task=config.task, input_dim=input_dim)
    logger.info(f"Available models: {list(registry.keys())}")
    
    # Validate models
    valid_models = [m for m in config.selected_models if m in registry]
    if not valid_models:
        logger.error("No valid models selected!")
        return
    
    use_cv = config.folds > 1
    
    if use_cv:
        logger.info(f"\nK-Fold Cross-Validation: {config.folds} folds")
        logger.info(f"\n--- Stage 1: External Test Set Split ---")
        logger.info(f"Using {config.split_method} split")
        
        if config.split_method == "scaffold":
            dev_indices, ext_test_indices = scaffold_split(smiles_list, y, config.test_size, config.seed)
            dev_scaffolds = set([get_scaffold(smiles_list[i]) for i in dev_indices])
            ext_test_scaffolds = set([get_scaffold(smiles_list[i]) for i in ext_test_indices])
            overlap = dev_scaffolds & ext_test_scaffolds
            logger.info(f"Development Set: {len(dev_indices)} samples, {len(dev_scaffolds)} unique scaffolds")
            logger.info(f"External Test Set: {len(ext_test_indices)} samples, {len(ext_test_scaffolds)} unique scaffolds")
            logger.info(f"Scaffold overlap: {len(overlap)} scaffolds")
        elif config.split_method == "stratified":
            if config.task == "classification":
                dev_indices, ext_test_indices = train_test_split(
                    np.arange(len(y)), y, test_size=config.test_size, random_state=config.seed, stratify=y
                )
            else:
                dev_indices, ext_test_indices = train_test_split(
                    np.arange(len(y)), test_size=config.test_size, random_state=config.seed
                )
            logger.info(f"Development Set: {len(dev_indices)} samples")
            logger.info(f"External Test Set: {len(ext_test_indices)} samples")
        else:
            dev_indices, ext_test_indices = train_test_split(
                np.arange(len(y)), test_size=config.test_size, random_state=config.seed
            )
            logger.info(f"Development Set: {len(dev_indices)} samples")
            logger.info(f"External Test Set: {len(ext_test_indices)} samples")
        
        y_dev = y[dev_indices]
        X_dev = X[dev_indices] if X is not None else None
        smiles_dev = [smiles_list[i] for i in dev_indices]
        ids_dev = [ids[i] for i in dev_indices]
        
        y_ext_test = y[ext_test_indices]
        X_ext_test = X[ext_test_indices] if X is not None else None
        smiles_ext_test = [smiles_list[i] for i in ext_test_indices]
        ids_ext_test = [ids[i] for i in ext_test_indices]
        
        logger.info(f"\n--- Stage 2: K-Fold CV on Development Set ---")
        
        if X_dev is not None:
            logger.info(f"\n--- Global Feature Filtering on Development Set ---")
            X_dev_filtered, dev_feature_mask, dev_feature_names_filtered = apply_global_feature_filtering(
                X_dev, config, logger, feature_names=fp_cols
            )
        else:
            X_dev_filtered, dev_feature_mask, dev_feature_names_filtered = X_dev, None, None
        
        if config.cv_split_method == "scaffold":
            logger.info("Using ScaffoldKFold with stratified round-robin allocation")
            kfold = ScaffoldKFold(n_splits=config.folds, shuffle=True, random_state=config.seed)
        else:
            logger.info("Using standard KFold/StratifiedKFold")
            if config.task == "classification":
                kfold = StratifiedKFold(n_splits=config.folds, shuffle=True, random_state=config.seed)
            else:
                kfold = KFold(n_splits=config.folds, shuffle=True, random_state=config.seed)
        
        all_fold_results = []
        
        for fold_idx, (train_indices, val_indices) in enumerate(kfold.split(smiles_dev, y_dev)):
            logger.info(f"\n{'='*60}")
            logger.info(f"Fold {fold_idx + 1}/{config.folds}")
            logger.info(f"{'='*60}")
            
            y_train, y_val = y_dev[train_indices], y_dev[val_indices]
            X_train, X_val = (X_dev_filtered[train_indices], X_dev_filtered[val_indices]) if X_dev_filtered is not None else (None, None)
            smiles_train = [smiles_dev[i] for i in train_indices]
            smiles_val = [smiles_dev[i] for i in val_indices]
            ids_train = [ids_dev[i] for i in train_indices]
            ids_val = [ids_dev[i] for i in val_indices]
            
            # Simple training (simplified for compact version)
            for model_key in valid_models:
                model_config = registry[model_key]
                result = train_model_wrapper(
                    model_key, model_config, X_train, y_train, X_val, y_val,
                    config, config.task, smiles_train, smiles_val, logger
                )
                if result is not None:
                    # Calculate metrics
                    if X_train is not None or 'MLP' in model_key:
                        y_pred_train = result['y_pred_train']
                        y_proba_train = result['y_proba_train']
                        metrics_train = calculate_metrics(y_train, y_pred_train, y_proba_train, config.task, config.ef_percentile)
                    else:
                        metrics_train = {}
                    
                    if X_val is not None and 'MLP' in model_key:
                        y_pred_val = result['y_pred_val']
                        y_proba_val = result['y_proba_val']
                        metrics_val = calculate_metrics(y_val, y_pred_val, y_proba_val, config.task, config.ef_percentile)
                    else:
                        metrics_val = {}
                    
                    fold_result = {
                        'fold': fold_idx + 1, 'model': model_key, 'model_type': model_config.get('type'),
                        **{f'{k}_train': v for k, v in metrics_train.items()},
                        **{f'{k}_val': v for k, v in metrics_val.items()}
                    }
                    all_fold_results.append(fold_result)
            
            if torch.cuda.is_available(): torch.cuda.empty_cache()
        
        # Save fold results
        fold_results_df = pd.DataFrame(all_fold_results)
        fold_results_df.to_csv(out_dir / "results" / "fold_results.csv", index=False)
        
        # Calculate summary
        model_cv_results = {}
        for result in all_fold_results:
            model_name = result['model']
            if model_name not in model_cv_results:
                model_cv_results[model_name] = []
            model_cv_results[model_name].append(result)
        
        cv_summary = []
        for model_name, fold_results_list in model_cv_results.items():
            summary_row = {'model': model_name, 'model_type': fold_results_list[0]['model_type']}
            metric_keys = [k for k in fold_results_list[0].keys() if k not in ['fold', 'model', 'model_type']]
            for metric_key in metric_keys:
                values = [r[metric_key] for r in fold_results_list]
                summary_row[f'{metric_key}_mean'] = np.mean(values)
                summary_row[f'{metric_key}_std'] = np.std(values)
            cv_summary.append(summary_row)
        
        cv_summary_df = pd.DataFrame(cv_summary)
        cv_summary_df.to_csv(out_dir / "results" / "cv_summary.csv", index=False)
        
        logger.info(f"\n{'='*60}")
        logger.info("CV Summary")
        logger.info(f"{'='*60}")
        for _, row in cv_summary_df.iterrows():
            logger.info(f"\n{row['model']}:")
            if config.task == "classification":
                logger.info(f"  AUC: {row.get('AUC_val_mean', 'N/A'):.4f} ± {row.get('AUC_val_std', 'N/A'):.4f}")
                logger.info(f"  PR-AUC: {row.get('PR_AUC_val_mean', 'N/A'):.4f} ± {row.get('PR_AUC_val_std', 'N/A'):.4f}")
                logger.info(f"  MCC: {row.get('MCC_val_mean', 'N/A'):.4f} ± {row.get('MCC_val_std', 'N/A'):.4f}")
            else:
                logger.info(f"  R2: {row.get('R2_val_mean', 'N/A'):.4f} ± {row.get('R2_val_std', 'N/A'):.4f}")
                logger.info(f"  RMSE: {row.get('RMSE_val_mean', 'N/A'):.4f} ± {row.get('RMSE_val_std', 'N/A'):.4f}")
        
        # Stage 3: External Test Set Evaluation
        logger.info(f"\n{'='*60}")
        logger.info("Stage 3: External Test Set Evaluation (Multi-Seed)")
        logger.info(f"{'='*60}")
        logger.info(f"Training all models on full Development Set ({len(y_dev)} samples)")
        logger.info(f"Evaluating all models on External Test Set ({len(y_ext_test)} samples)")
        logger.info(f"Using {len(config.seeds)} seeds: {config.seeds}")
        logger.info(f"Best model will be selected based on: {config.external_test_metric}")
        
        # Apply feature processing
        if X_dev is not None and X_ext_test is not None:
            logger.info(f"\nApplying feature processing to Development Set and External Test Set...")
            X_dev_for_ext_test = X_dev[:, dev_feature_mask] if dev_feature_mask is not None else X_dev
            X_ext_test_for_ext_test = X_ext_test[:, dev_feature_mask] if dev_feature_mask is not None else X_ext_test
            logger.info(f"Using same features as K-Fold CV: {X_dev_for_ext_test.shape[1]} features")
        else:
            X_dev_for_ext_test, X_ext_test_for_ext_test = X_dev, X_ext_test
        
        all_ext_test_results = []
        all_ext_test_predictions = []
        trained_models = {}
        
        # Evaluate each model on External Test Set
        for model_key in valid_models:
            model_config = registry[model_key]
            model_type_category = 'deep_learning' if model_config['type'] in ['pytorch', 'pytorch_geometric', 'transformer'] else 'traditional'
            
            logger.info(f"\n{'='*60}")
            logger.info(f"Model: {model_key}")
            logger.info(f"{'='*60}")
            
            model_seed_results = []
            
            for seed_idx, seed in enumerate(config.seeds):
                logger.info(f"\n--- Seed {seed_idx+1}/{len(config.seeds)} (seed={seed}) ---")
                set_all_seeds(seed, logger)
                
                # Prepare data
                if model_config['type'] in ['sklearn']:
                    model_preprocessing_type = get_model_type(model_key)
                    if X_dev_for_ext_test is not None and X_ext_test_for_ext_test is not None:
                        if model_preprocessing_type == 'gradient_based':
                            from sklearn.preprocessing import StandardScaler
                            scaler = StandardScaler()
                            X_train_final = scaler.fit_transform(X_dev_for_ext_test)
                            X_test_final = scaler.transform(X_ext_test_for_ext_test)
                            logger.info(f"  Standardization applied for {model_key}")
                        else:
                            X_train_final, X_test_final = X_dev_for_ext_test, X_ext_test_for_ext_test
                            logger.info(f"  Standardization skipped for {model_key} (tree-based)")
                    else:
                        X_train_final, X_test_final = None, None
                else:
                    if model_config['type'] == 'pytorch':
                        if X_dev_for_ext_test is not None and X_ext_test_for_ext_test is not None:
                            from sklearn.preprocessing import StandardScaler
                            scaler = StandardScaler()
                            X_train_final = scaler.fit_transform(X_dev_for_ext_test)
                            X_test_final = scaler.transform(X_ext_test_for_ext_test)
                            logger.info(f"  Standardization applied for {model_key}")
                        else:
                            X_train_final, X_test_final = None, None
                    else:
                        X_train_final, X_test_final = None, None
                
                # Train on full Development Set
                model_result = train_model_wrapper(
                    model_key, model_config, X_train_final, y_dev, None, y_dev,
                    config, config.task, smiles_dev, None, logger
                )
                
                if model_result is None:
                    logger.warning(f"  Failed to train {model_key} with seed {seed}")
                    continue
                
                # Store model (last seed)
                if seed_idx == len(config.seeds) - 1:
                    trained_models[model_key] = model_result['model']
                
                logger.info(f"  ✓ {model_key} trained on full Development Set (seed={seed})")
                
                # Evaluate on External Test Set
                logger.info(f"  --- Evaluating on External Test Set (seed={seed}) ---")
                
                if model_config['type'] in ['pytorch_geometric', 'transformer', 'pytorch']:
                    if model_config['type'] == 'pytorch_geometric':
                        test_dataset = MoleculeDataset(smiles_ext_test, y_ext_test)
                        test_loader = GeometricDataLoader(test_dataset, batch_size=config.batch_size, shuffle=False)
                    elif model_config['type'] == 'transformer':
                        model = model_result['model']
                        test_dataset = ChemBERTaDataset(smiles_ext_test, y_ext_test, model.tokenizer)
                        test_loader = DataLoader(test_dataset, batch_size=config.batch_size, shuffle=False, collate_fn=chemberta_collate_fn)
                    else:
                        test_dataset = TensorDataset(torch.tensor(X_test_final, dtype=torch.float32), torch.tensor(y_ext_test, dtype=torch.float32))
                        test_loader = DataLoader(test_dataset, batch_size=config.batch_size, shuffle=False)
                    
                    y_pred_ext_test, y_proba_ext_test = predict_pytorch_model(model_result['model'], test_loader, config.task)
                else:
                    if hasattr(model_result['model'], 'predict_proba'):
                        y_proba_ext_test = model_result['model'].predict_proba(X_test_final)[:, 1]
                    elif hasattr(model_result['model'], 'decision_function'):
                        y_proba_ext_test = model_result['model'].decision_function(X_test_final)
                        if config.task == "classification":
                            y_proba_ext_test = 1.0 / (1.0 + np.exp(-y_proba_ext_test))
                    else:
                        y_proba_ext_test = model_result['model'].predict(X_test_final)
                    y_pred_ext_test = (y_proba_ext_test >= 0.5).astype(int) if config.task == "classification" else y_proba_ext_test
                
                # Calculate metrics
                ext_test_metrics = calculate_metrics(y_ext_test, y_pred_ext_test, y_proba_ext_test, config.task, config.ef_percentile)
                
                seed_result = {
                    'model': model_key, 'model_type': model_type_category, 'seed': seed,
                    'n_ext_test_samples': len(y_ext_test), **ext_test_metrics
                }
                model_seed_results.append(seed_result)
                
                # Store predictions
                pred_df = pd.DataFrame({
                    'model': model_key, 'seed': seed, 'id': ids_ext_test, 'smiles': smiles_ext_test,
                    'true_label': y_ext_test, 'predicted_label': y_pred_ext_test, 'predicted_probability': y_proba_ext_test
                })
                all_ext_test_predictions.append(pred_df)
                
                # Log results
                if config.task == "classification":
                    logger.info(f"  Seed {seed} Results: AUC={ext_test_metrics.get('AUC', 'N/A'):.4f}, PR-AUC={ext_test_metrics.get('PR_AUC', 'N/A'):.4f}, MCC={ext_test_metrics.get('MCC', 'N/A'):.4f}")
                else:
                    logger.info(f"  Seed {seed} Results: R2={ext_test_metrics.get('R2', 'N/A'):.4f}, RMSE={ext_test_metrics.get('RMSE', 'N/A'):.4f}")
                
                if model_type_category == 'deep_learning' and torch.cuda.is_available():
                    torch.cuda.empty_cache()
            
            if not model_seed_results:
                logger.warning(f"{model_key} failed on all seeds")
                continue
            
            all_ext_test_results.extend(model_seed_results)
            
            # Calculate mean and std
            logger.info(f"\n--- {model_key} Multi-Seed Summary ---")
            metric_keys = [k for k in model_seed_results[0].keys() if k not in ['model', 'model_type', 'seed', 'n_ext_test_samples']]
            for metric_key in metric_keys:
                values = [r[metric_key] for r in model_seed_results]
                logger.info(f"  {metric_key}: {np.mean(values):.4f} ± {np.std(values):.4f}")
        
        # Save External Test Set results
        ext_test_results_df = pd.DataFrame(all_ext_test_results)
        ext_test_results_df.to_csv(out_dir / "results" / "external_test_results.csv", index=False)
        
        ext_test_predictions_df = pd.concat(all_ext_test_predictions, ignore_index=True)
        ext_test_predictions_df.to_csv(out_dir / "predictions" / "external_test_predictions.csv", index=False)
        
        # Create summary
        summary_results = []
        for model_name in [r['model'] for r in all_ext_test_results]:
            model_seed_results_list = [r for r in all_ext_test_results if r['model'] == model_name]
            if not model_seed_results_list: continue
            
            summary_row = {'model': model_name, 'model_type': model_seed_results_list[0]['model_type'], 'num_seeds': len(model_seed_results_list)}
            metric_keys = [k for k in model_seed_results_list[0].keys() if k not in ['model', 'model_type', 'seed', 'n_ext_test_samples']]
            for metric_key in metric_keys:
                values = [r[metric_key] for r in model_seed_results_list]
                summary_row[f'{metric_key}_mean'] = np.mean(values)
                summary_row[f'{metric_key}_std'] = np.std(values)
            summary_results.append(summary_row)
        
        summary_df = pd.DataFrame(summary_results)
        summary_df.to_csv(out_dir / "results" / "external_test_summary.csv", index=False)
        
        # Find best models
        logger.info(f"\n{'='*60}")
        logger.info("External Test Set Results Summary")
        logger.info(f"{'='*60}")
        
        ext_test_metric_key = f'{config.external_test_metric}_mean'
        traditional_summary = [r for r in summary_results if r['model_type'] == 'traditional']
        deep_learning_summary = [r for r in summary_results if r['model_type'] == 'deep_learning']
        
        best_traditional = None
        best_deep_learning = None
        
        if traditional_summary:
            best_traditional = min(traditional_summary, key=lambda x: x.get(ext_test_metric_key, -float('inf')))
            logger.info(f"\nBest Traditional Model: {best_traditional['model']} ({config.external_test_metric}={best_traditional.get(ext_test_metric_key, 'N/A'):.4f})")
        
        if deep_learning_summary:
            best_deep_learning = max(deep_learning_summary, key=lambda x: x.get(ext_test_metric_key, -float('inf')))
            logger.info(f"\nBest Deep Learning Model: {best_deep_learning['model']} ({config.external_test_metric}={best_deep_learning.get(ext_test_metric_key, 'N/A'):.4f})")
        
        # Save best models
        if best_traditional or best_deep_learning:
            best_model_dir = out_dir / "models" / "best_models"
            best_model_dir.mkdir(parents=True, exist_ok=True)
            
            logger.info(f"\n{'='*60}")
            logger.info("Saving Best Models")
            logger.info(f"{'='*60}")
            
            if best_traditional is not None:
                model_key = best_traditional['model']
                if model_key in trained_models:
                    model_path = best_model_dir / f"best_traditional_{model_key}"
                    _save_sklearn_model(trained_models[model_key], model_path)
                    logger.info(f"  ✓ Saved best traditional model: {model_key}")
            
            if best_deep_learning is not None:
                model_key = best_deep_learning['model']
                if model_key in trained_models:
                    model_path = best_model_dir / f"best_deep_learning_{model_key}"
                    model_config = registry.get(model_key)
                    if model_config:
                        model_type = model_config.get('type', 'pytorch')
                        _save_pytorch_model(trained_models[model_key], model_path, model_type)
                        logger.info(f"  ✓ Saved best deep learning model: {model_key}")
        
        # Save feature processors
        if dev_feature_mask is not None:
            processor_dir = out_dir / "feature_processors"
            np.save(processor_dir / "feature_mask.npy", dev_feature_mask)
            if dev_feature_names_filtered is not None:
                with open(processor_dir / "feature_names.json", 'w') as f:
                    json.dump(dev_feature_names_filtered, f)
        
        logger.info(f"\nExternal Test Set results saved to:")
        logger.info(f"  - {out_dir / 'results' / 'external_test_results.csv'}")
        logger.info(f"  - {out_dir / 'results' / 'external_test_summary.csv'}")
        logger.info(f"  - {out_dir / 'predictions' / 'external_test_predictions.csv'}")
    
    else:
        logger.info(f"\n{'='*60}")
        logger.info("Single Train/Validation Split")
        logger.info(f"{'='*60}")
        
        # Single split based on method
        if config.split_method == "scaffold":
            if smiles_list is None:
                logger.error("SMILES data required for scaffold split!")
                return
            logger.info("Using scaffold-based split")
            train_indices, val_indices = scaffold_split(smiles_list, y, config.test_size, config.seed)
            
            # Apply split
            y_train, y_val = y[train_indices], y[val_indices]
            if X is not None:
                X_train, X_val = X[train_indices], X[val_indices]
            else:
                X_train, X_val = None, None
            smiles_train = [smiles_list[i] for i in train_indices]
            smiles_val = [smiles_list[i] for i in val_indices]
            if ids:
                ids_train = [ids[i] for i in train_indices]
                ids_val = [ids[i] for i in val_indices]
            else:
                ids_train, ids_val = None, None
            
            # Count scaffolds
            train_scaffolds = set([get_scaffold(s) for s in smiles_train])
            val_scaffolds = set([get_scaffold(s) for s in smiles_val])
            overlap = train_scaffolds & val_scaffolds
            logger.info(f"Train scaffolds: {len(train_scaffolds)}, Val scaffolds: {len(val_scaffolds)}, Overlap: {len(overlap)}")
        
        elif config.split_method == "stratified":
            logger.info("Using stratified split")
            if config.task == "classification":
                train_indices, val_indices = train_test_split(np.arange(len(y)), test_size=config.test_size, random_state=config.seed, stratify=y)
            else:
                train_indices, val_indices = train_test_split(np.arange(len(y)), test_size=config.test_size, random_state=config.seed)
            
            # Apply split
            y_train, y_val = y[train_indices], y[val_indices]
            if X is not None:
                X_train, X_val = X[train_indices], X[val_indices]
            else:
                X_train, X_val = None, None
            smiles_train = [smiles_list[i] for i in train_indices]
            smiles_val = [smiles_list[i] for i in val_indices]
            if ids:
                ids_train = [ids[i] for i in train_indices]
                ids_val = [ids[i] for i in val_indices]
            else:
                ids_train, ids_val = None, None
        
        else:
            logger.info("Using random split")
            train_indices, val_indices = train_test_split(np.arange(len(y)), test_size=config.test_size, random_state=config.seed)
            
            # Apply split
            y_train, y_val = y[train_indices], y[val_indices]
            if X is not None:
                X_train, X_val = X[train_indices], X[val_indices]
            else:
                X_train, X_val = None, None
            smiles_train = [smiles_list[i] for i in train_indices]
            smiles_val = [smiles_list[i] for i in val_indices]
            if ids:
                ids_train = [ids[i] for i in train_indices]
                ids_val = [ids[i] for i in val_indices]
            else:
                ids_train, ids_val = None, None
        
        # Train models with multi-seed support
        model_save_dir = out_dir / "models" / "seed_models"
        model_save_dir.mkdir(parents=True, exist_ok=True)
        
        all_fold_results = train_single_fold(
            config=config, X_train=X_train, y_train=y_train, X_val=X_val, y_val=y_val,
            smiles_train=smiles_train, smiles_val=smiles_val, ids_train=ids_train, ids_val=ids_val,
            registry=registry, fold_idx=None, save_models=True, model_save_dir=model_save_dir, logger=logger
        )
        
        # Save results
        summary_df = pd.DataFrame(all_fold_results)
        summary_df.to_csv(out_dir / "results" / "summary_metrics.csv", index=False)
        
        # Find and save best models
        if config.task == "classification":
            metric = 'AUC_val_mean'
        else:
            metric = 'R2_val_mean'
        
        traditional_results = [r for r in all_fold_results if r['model_type'] == 'traditional']
        deep_learning_results = [r for r in all_fold_results if r['model_type'] == 'deep_learning']
        
        best_traditional = None
        best_deep_learning = None
        
        if traditional_results:
            best_traditional = max(traditional_results, key=lambda x: x.get(metric, -float('inf')))
            logger.info(f"\nBest Traditional Model: {best_traditional['model']} ({metric}={best_traditional.get(metric, 'N/A'):.4f})")
        
        if deep_learning_results:
            best_deep_learning = max(deep_learning_results, key=lambda x: x.get(metric, -float('inf')))
            logger.info(f"\nBest Deep Learning Model: {best_deep_learning['model']} ({metric}={best_deep_learning.get(metric, 'N/A'):.4f})")
        
        # Save best models
        best_model_dir = out_dir / "models" / "best_models"
        best_model_dir.mkdir(parents=True, exist_ok=True)
        
        # Note: In single split mode with multi-seed, models are already saved in seed_models
        # We could copy the best ones to best_models, but for simplicity we'll just log them
        
        logger.info(f"\nResults saved to: {out_dir / 'results' / 'summary_metrics.csv'}")

if __name__ == "__main__":
    main_cli()
